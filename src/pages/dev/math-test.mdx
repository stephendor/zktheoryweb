### Python — Vietoris-Rips Complex

```python
import numpy as np
from scipy.spatial.distance import cdist

def rips_complex(points: np.ndarray, epsilon: float) -> list[tuple]:
    """Return edges of the Vietoris-Rips complex at scale epsilon."""
    dists = cdist(points, points)
    n = len(points)
    return [
        (i, j)
        for i in range(n)
        for j in range(i + 1, n)
        if dists[i, j] <= epsilon
    ]
```

### TypeScript — Persistence Diagram

```typescript
interface PersistencePoint {
  birth: number;
  death: number;
  dimension: number;
}

function bottleneckDistance(d1: PersistencePoint[], d2: PersistencePoint[]): number {
  // Naive O(n²) matching for illustration — use a proper assignment solver in production
  let maxDist = 0;
  for (const p of d1) {
    const nearest = d2.reduce((best, q) => {
      const dist = Math.max(Math.abs(p.birth - q.birth), Math.abs(p.death - q.death));
      return dist < best ? dist : best;
    }, Infinity);
    maxDist = Math.max(maxDist, nearest);
  }
  return maxDist;
}
```
